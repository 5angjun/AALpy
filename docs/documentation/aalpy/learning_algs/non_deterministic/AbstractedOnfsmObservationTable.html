 56    def abstract_obs_table(self):
 57        """
 58        Creation of abstracted observation table. The provided abstraction mapping is used to
 59        replace outputs by abstracted outputs.
 60        """
 61
 62        self.S = self.observation_table.S
 63        self.S_dot_A = list(set(self.observation_table.get_extended_S()).union(set(self.S_dot_A) - set(self.S)))
 64        self.E = self.observation_table.E
 65
 66        update_S = self.S + self.S_dot_A
 67        update_E = self.E
 68
 69        for s in update_S:
 70            for e in update_E:
 71                for o_tup in self.get_all_outputs(s, e):
 72                    abstracted_outputs = []
 73                    o_tup = tuple([o_tup])
 74                    for outputs in o_tup:
 75                        for o in outputs:
 76                            abstract_output = self.get_abstraction(o)
 77                            abstracted_outputs.append(abstract_output)
 78                    self.add_to_T(s, e, tuple(abstracted_outputs))

285    def gen_hypothesis(self) -> Onfsm:
286        """
287        Generate automaton based on the values found in the abstracted observation table.
288
289        Returns:
290
291            Current abstracted hypothesis
292
293        """
294        state_distinguish = dict()
295        states_dict = dict()
296        initial = None
297
298        unified_S = self.S + self.S_dot_A
299
300        stateCounter = 0
301        for prefix in self.S:
302            state_id = f's{stateCounter}'
303            states_dict[prefix] = OnfsmState(state_id)
304
305            states_dict[prefix].prefix = prefix
306            state_distinguish[self.row_to_hashable(prefix)] = states_dict[prefix]
307
308            if prefix == self.S[0]:
309                initial = states_dict[prefix]
310            stateCounter += 1
311
312        for prefix in self.S:
313            similar_rows = []
314            for row in unified_S:
315                if self.row_to_hashable(row) == self.row_to_hashable(prefix):
316                    similar_rows.append(row)
317            for row in similar_rows:
318                for a in self.A:
319                    for t in self.get_all_outputs(row, a):
320                        s_entry = (row[0] + a, row[1] + t)
321                        if s_entry in unified_S:
322                            state_in_S = state_distinguish[self.row_to_hashable(s_entry)]
323
324                            if (t[0], state_in_S) not in states_dict[prefix].transitions[a[0]]:
325                                states_dict[prefix].transitions[a[0]].append((t[0], state_in_S))
326
327        assert initial
328        automaton = Onfsm(initial, [s for s in states_dict.values()])
329        automaton.characterization_set = self.E
330
331        return automaton

367    def cex_processing(self, cex: tuple, hypothesis: Onfsm):
368        """
369        Add counterexample to the observation table. If the counterexample leads to a state where an output of the
370        same equivalence class already exists, the prefixes of the counterexample are added to S.A.
371        Otherwise, the postfixes of counterexample are added to E.
372
373
374        Args:
375
376            cex: counterexample that should be added to the observation table
377            hypothesis: onfsm that implements the counterexample
378        """
379
380        cex_len = len(cex[0])
381        hypothesis.reset_to_initial()
382
383        for step in range(0, cex_len - 1):
384            hypothesis.step_to(cex[0][step], cex[1][step])
385
386        possible_outputs = hypothesis.outputs_on_input(cex[0][cex_len - 1])
387
388        equivalent_output = False
389
390        for out in possible_outputs:
391            abstracted_out = self.get_abstraction(out)
392            abstracted_out_cex = self.get_abstraction(cex[1][cex_len - 1])
393            if abstracted_out_cex == abstracted_out:
394                equivalent_output = True
395                break
396
397        if equivalent_output:
398            # add prefixes of cex to S_dot_A
399            cex_prefixes = [(tuple(cex[0][0:i + 1]), tuple(cex[1][0:i + 1])) for i in range(0, len(cex[0]))]
400            prefixes_to_extend = self.extend_S_dot_A(cex_prefixes)
401
402            # CHANGED: REMOVED
403            # self.observation_table.S_dot_A.extend(prefixes_to_extend)
404            self.update_obs_table(s_set=prefixes_to_extend)
405        else:
406            # add distinguishing suffixes of cex to E
407            # CHANGED CEX PROX
408            # TODO: this will now not work as cex processing was changed
409            # cex_suffixes = non_det_longest_prefix_cex_processing(self.observation_table, cex)
410            cex_suffixes = all_suffixes(cex[0])
411
412            added_suffixes = extend_set(self.observation_table.E, cex_suffixes)
413            self.update_obs_table(e_set=added_suffixes)
