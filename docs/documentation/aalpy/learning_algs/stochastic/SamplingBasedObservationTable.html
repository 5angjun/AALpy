 53    def refine_not_completed_cells(self, n_resample, uniform=False):
 54        """
 55        Firstly a prefix-tree acceptor is constructed for all non-completed cells and then that tree is used
 56        for online testing/sampling.
 57
 58        Args:
 59
 60          uniform: if true, all cells will be uniformly sampled (Default value = False)
 61          n_resample: Number of resamples
 62
 63        Returns:
 64
 65            False if no cells are to be refined, True if refining happened
 66        """
 67        if self.automaton_type == 'mdp':
 68            pta_root = Node(self.initial_output[0])
 69        else:
 70            pta_root = Node(None)
 71
 72        dynamic = 0
 73        if self.strategy == 'classic':
 74            to_refine = []
 75            for s in self.S + list(self.get_extended_s()):
 76                for e in self.E:
 77                    if not self.teacher.complete_query(s, e):
 78                        to_refine.append(s + e)
 79
 80            if not to_refine:
 81                return False
 82
 83            to_refine.sort(key=len, reverse=True)
 84
 85            for trace in to_refine:
 86                self.add_to_PTA(pta_root, trace)
 87
 88        else:
 89            for s in self.S + list(self.get_extended_s()):
 90                if uniform:
 91                    for e in self.E:
 92                        self.add_to_PTA(pta_root, s + e, 1)
 93                else:
 94                    for e in self.E:
 95                        longest_row_trace_prefix = (s + e)[:-1]
 96                        while longest_row_trace_prefix not in self.T.keys():
 97                            longest_row_trace_prefix = longest_row_trace_prefix[:-1]
 98                        row_repr = 0
 99                        for r in self.compatibility_classes_representatives:
100                            if self.are_rows_compatible(longest_row_trace_prefix, r):
101                                row_repr += 1
102                        # row_repr can be zero for non-closed
103                        # (int(row_repr - 1 * 2))
104                        uncertainty_value = max((row_repr - 1) * 2, 1)
105                        dynamic += uncertainty_value
106                        self.add_to_PTA(pta_root, s + e, uncertainty_value)
107
108        resample_value = n_resample if self.strategy == 'classic' else max(dynamic // 2, 500)
109
110        for i in range(resample_value):
111            self.teacher.tree_query(pta_root)
112        return True

157    def make_closed_and_consistent(self):
158        """
159        Observation table is updated until it is closed and consistent. Note that due the updated notion of row
160        equivalence no sampling is needed.
161        """
162        self.update_compatibility_classes()
163
164        while True:
165            closed, consistent = False, False
166            row_to_close = self.get_row_to_close()
167            if not row_to_close:
168                closed = True
169            if row_to_close:

172                self.update_compatibility_classes()

174            consistency_violation = self.get_consistency_violation()
175            if not consistency_violation:
176                consistent = True
177            if consistency_violation:
178                if consistency_violation not in self.E:
179                    self.E.append(consistency_violation)
180                self.update_obs_table_with_freq_obs()
181                self.update_compatibility_classes()
182
183            if closed and consistent:

204    def get_consistency_violation(self, ignore=None):
205        """Find and return cause of consistency violation. Only computed on the compatibility class representatives.
206        :return: element of input + element of output + element of e that lead to the inconsistency
207
208        Args:
209          ignore:  (Default value = None)
210
211        Returns:
212
213            i + o + e that violate consistency
214        """
215        if self.cex_processing is not None:
216            return None
217
218        for ind, s1 in enumerate(self.S):
219            for s2 in self.S[ind + 1:]:
220                if self.are_rows_compatible(s1, s2, ignore):
221                    i_o_pairs = [(i, tuple([o])) for i in self.input_alphabet for o in self.T[s1][i].keys()]
222                    for i, o in i_o_pairs:
223                        s1_keys = self.T[s1 + i + o].keys()
224                        s2_keys = self.T[s2 + i + o].keys()
225                        if not s1_keys or not s2_keys:
226                            continue
227
228                        for e in self.E:
229                            if e == ignore:
230                                continue
231                            if self.are_cells_incompatible(s1 + i + o, s2 + i + o, e):
232                                return i + o + e

235    def get_representative(self, target):
236        """
237
238        Args:
239          target: row in the observation table
240
241        Returns:
242          a representative compatible with the target
243
244        """
245        if self.compatibility_checker.use_diff_value():
246            smallest_diff_value = 2 ** 32
247            best_rep = None
248            if target in self.compatibility_classes_representatives:
249                return target
250            for r in self.compatibility_classes_representatives:
251                if self.automaton_type == "mdp" and r[-1] != target[-1]:
252                    continue
253                if not self.are_rows_compatible(r, target):
254                    continue
255                diff_value = 0
256                row_target = self.T[target]
257                row_r = self.T[r]
258                for e in self.E:
259                    diff_value += self.compatibility_checker.difference_value(row_r.get(e, None),
260                                                                              row_target.get(e, None))
261                if diff_value < smallest_diff_value:
262                    # if smallest_diff_value != 2**32:
263                    #    print("Found a better rep")
264                    smallest_diff_value = diff_value
265                    best_rep = r
266            return best_rep
267        else:
268            if target in self.S:
269                for r in self.compatibility_classes_representatives:
270                    if target == r or target in self.compatibility_class[r]:
271                        return r
272            else:
273                for r in self.compatibility_classes_representatives:
274                    if self.are_rows_compatible(r, target):
275                        return r
276        assert False

361    def stop(self, learning_round, chaos_cex_present, cex, stopping_range_dict, min_rounds=10, max_rounds=None,
362             target_unambiguity=0.99, print_unambiguity=False):
363        """
364        Decide if learning should terminate.
365
366        Args:
367
368          learning_round: current learning round
369          chaos_cex_present: is chaos counterexample present in the hypothesis
370          cex: counterexample found by the eq oracle
371          stopping_range_dict: dictionary where keys are number of last unambiguity values and value is
372          maximum differance allowed between them
373          min_rounds: minimum number of learning rounds (Default value = 5)
374          max_rounds: maximum number of learning rounds (Default value = None)
375          target_unambiguity: percentage of rows with unambiguous representatives (Default value = 0.99)
376          print_unambiguity: if true, current unambiguity rate will be printed (Default value = False)
377
378        Returns:
379
380          True if stopping condition satisfied, false otherwise
381        """
382        if max_rounds:
383            assert min_rounds <= max_rounds
384        if max_rounds and learning_round == max_rounds:
385            return True
386        if chaos_cex_present or cex is not None:
387            return False
388
389        extended_s = list(self.get_extended_s())
390        self.update_compatibility_classes()
391        numerator = 0
392        for row in self.S + extended_s:
393            row_repr = 0
394            for r in self.compatibility_classes_representatives:
395                if self.are_rows_compatible(row, r):
396                    row_repr += 1
397            numerator += 1 if row_repr == 1 else 0
398
399        unambiguous_rows_percentage = numerator / len(self.S + extended_s)
400
401        self.unambiguity_values.append(unambiguous_rows_percentage)
402        if self.strategy != 'classic' and learning_round >= min_rounds:
403            # keys are number of last unambiguity values and value is maximum differance allowed between them
404
405            for num_last, diff in stopping_range_dict.items():
406                if len(self.unambiguity_values) < num_last:
407                    continue
408                last_n_unamb = self.unambiguity_values[-num_last:]
409                if abs(max(last_n_unamb) - min(last_n_unamb) <= diff):
410                    return True
411
412        if print_unambiguity and learning_round % 5 == 0:
413            print(f'Unambiguous rows: {round(unambiguous_rows_percentage * 100, 2)}%;'
414                  f' {numerator} out of {len(self.S + extended_s)}')
415        if learning_round >= min_rounds and unambiguous_rows_percentage >= target_unambiguity:
416            return True
417
418        return False

434    def are_cells_incompatible(self, s1, s2, e):
435        """
436        Checks if 2 cells are considered different.
437
438        Args:
439
440          s1: prefix of row s1
441          s2: prefix of row s2
442          e: element of E
443
444        Returns:
445
446          True if cells are different, false otherwise
447
448        """
449        if self.strategy == 'classic':
450            if self.teacher.complete_query(s1, e) and self.teacher.complete_query(s2, e):
451                return self.compatibility_checker.are_cells_different(self.T[s1][e], self.T[s2][e])

453            if e in self.T[s1] and e in self.T[s2]:
454                return self.compatibility_checker.are_cells_different(self.T[s1][e], self.T[s2][e])
455        else:

457                return self.compatibility_checker.are_cells_different(self.T[s1][e], self.T[s2][e], s1=s1, s2=s2, e=e)

460    def are_rows_compatible(self, s1, s2, e_ignore=None):
461        """
462        Check if the rows are compatible.
463        Rows are compatible if all cells are compatible(not different) and their prefixes
464        end in the same output element.

466        Args:
467          s1: prefix of row s1

485    def update_compatibility_classes(self):
486        """Updates the compatibility classes and stores their representatives."""
487        self.compatibility_class.clear()
488
489        class_rank_pair = []
490        for s in self.S:
491            rank = sum([sum(self.T[s][i].values()) for i in self.input_alphabet])
492            class_rank_pair.append((s, rank))
493
494        # sort according to frequency
495        class_rank_pair.sort(key=lambda x: x[1], reverse=True)
496
497        # # sort according to prefix length, and elements of same length sort by value
498        # class_rank_pair = [(s, -rank) for (s, rank) in class_rank_pair]
499        # class_rank_pair.sort(key=lambda x: (len(x[0]), x[1]))
500        # class_rank_pair = [(s, -rank) for (s, rank) in class_rank_pair]
501
502        compatibility_classes = [c[0] for c in class_rank_pair]
503
504        tmp_classes = list(compatibility_classes)
505        not_partitioned = list(self.S)
506
507        representatives = []
508        while not_partitioned:
509            r = tmp_classes.pop(0)
510            not_partitioned.remove(r)
511
512            cg_r = [s for s in not_partitioned if self.are_rows_compatible(r, s)]
513
514            self.compatibility_class[r] = cg_r
515
516            representatives.append(r)
517            for sp in cg_r:
518                not_partitioned.remove(sp)
519                tmp_classes.remove(sp)
520
521        self.compatibility_classes_representatives = representatives

553    def add_to_PTA(self, pta_root, trace, uncertainty_value=None):
554        """Adds a trace to the PTA. PTA is later used for online sampling. The uncertainty value is added to inputs as
555        frequencies, which specify how often a particular input should be sampled.
556
557        Args:
558          pta_root: root of the prefix tree acceptor
559          trace: trace to add to the PTA
560          uncertainty_value: uncertainty value (Default value = None)
561
562        Returns:
563
564        """
565        curr_node = pta_root
566        start = 1 if self.automaton_type == 'mdp' else 0
567        for index in range(start, len(trace), 2):
568            inp = trace[index]
569            if uncertainty_value:
570                # use frequencies for uncertainties
571                curr_node.input_frequencies[inp] += uncertainty_value
572            # need to add a dummy output in the leaves
573            output = trace[index + 1] if index + 1 < len(trace) else "dummy"
574            child = curr_node.get_child(inp, output)
575            if child:
576                curr_node = child
577            else:
578                new_node = Node(output)
579                curr_node.children[inp][output] = new_node
580                curr_node = new_node

 54    def refine_not_completed_cells(self, n_resample, uniform=False):
 55        """
 56        Firstly a prefix-tree acceptor is constructed for all non-completed cells and then that tree is used
 57        for online testing/sampling.
 58
 59        Args:
 60
 61          uniform: if true, all cells will be uniformly sampled (Default value = False)
 62          n_resample: Number of resamples
 63
 64        Returns:
 65
 66            False if no cells are to be refined, True if refining happened
 67        """
 68        if self.automaton_type == 'mdp':
 69            pta_root = Node(self.initial_output[0])
 70        else:
 71            pta_root = Node(None)
 72
 73        dynamic = 0
 74        if self.strategy == 'classic':
 75            to_refine = []
 76            for s in self.S + list(self.get_extended_s()):
 77                for e in self.E:
 78                    if not self.teacher.complete_query(s, e):
 79                        to_refine.append(s + e)
 80
 81            if not to_refine:
 82                return False
 83
 84            to_refine.sort(key=len, reverse=True)
 85
 86            for trace in to_refine:
 87                self.add_to_PTA(pta_root, trace)
 88
 89        else:
 90            for s in self.S + list(self.get_extended_s()):
 91                if uniform:
 92                    for e in self.E:
 93                        self.add_to_PTA(pta_root, s + e, 1)

236    def get_representative(self, target):
237        """
238
239        Args:
240          target: row in the observation table
241
242        Returns:
243          a representative compatible with the target
244
245        """
246        if self.compatibility_checker.use_diff_value():

362    def stop(self, learning_round, chaos_cex_present, cex, stopping_range_dict, min_rounds=10, max_rounds=None,
363             target_unambiguity=0.99, print_unambiguity=False):
364        """
365        Decide if learning should terminate.
366
367        Args:
368
369          learning_round: current learning round
370          chaos_cex_present: is chaos counterexample present in the hypothesis
371          cex: counterexample found by the eq oracle
372          stopping_range_dict: dictionary where keys are number of last unambiguity values and value is
373          maximum differance allowed between them
374          min_rounds: minimum number of learning rounds (Default value = 5)
375          max_rounds: maximum number of learning rounds (Default value = None)
376          target_unambiguity: percentage of rows with unambiguous representatives (Default value = 0.99)
377          print_unambiguity: if true, current unambiguity rate will be printed (Default value = False)
378
379        Returns:
380
381          True if stopping condition satisfied, false otherwise
382        """
383        if max_rounds:
384            assert min_rounds <= max_rounds
385        if max_rounds and learning_round == max_rounds:
386            return True
387        if chaos_cex_present or cex is not None:
388            return False
389
390        extended_s = list(self.get_extended_s())
391        self.update_compatibility_classes()
392        numerator = 0
393        for row in self.S + extended_s:
394            row_repr = 0
395            for r in self.compatibility_classes_representatives:
396                if self.are_rows_compatible(row, r):
397                    row_repr += 1
398            numerator += 1 if row_repr == 1 else 0
399
400        unambiguous_rows_percentage = numerator / len(self.S + extended_s)
401
402        self.unambiguity_values.append(unambiguous_rows_percentage)
403        if self.strategy != 'classic' and learning_round >= min_rounds:
404            # keys are number of last unambiguity values and value is maximum differance allowed between them
405
406            for num_last, diff in stopping_range_dict.items():
407                if len(self.unambiguity_values) < num_last:
408                    continue
409                last_n_unamb = self.unambiguity_values[-num_last:]
410                if abs(max(last_n_unamb) - min(last_n_unamb) <= diff):
411                    return True
412
413        if print_unambiguity and learning_round % 5 == 0:
414            print(f'Unambiguous rows: {round(unambiguous_rows_percentage * 100, 2)}%;'
415                  f' {numerator} out of {len(self.S + extended_s)}')
416        if learning_round >= min_rounds and unambiguous_rows_percentage >= target_unambiguity:
417            return True
418
419        return False

554    def add_to_PTA(self, pta_root, trace, uncertainty_value=None):
555        """Adds a trace to the PTA. PTA is later used for online sampling. The uncertainty value is added to inputs as
556        frequencies, which specify how often a particular input should be sampled.
557
558        Args:
559          pta_root: root of the prefix tree acceptor
560          trace: trace to add to the PTA
561          uncertainty_value: uncertainty value (Default value = None)
562
563        Returns:
564
565        """
566        curr_node = pta_root
567        start = 1 if self.automaton_type == 'mdp' else 0
568        for index in range(start, len(trace), 2):
569            inp = trace[index]
570            if uncertainty_value:
571                # use frequencies for uncertainties
572                curr_node.input_frequencies[inp] += uncertainty_value
573            # need to add a dummy output in the leaves
574            output = trace[index + 1] if index + 1 < len(trace) else "dummy"
575            child = curr_node.get_child(inp, output)
576            if child:
577                curr_node = child
578            else:
579                new_node = Node(output)
580                curr_node.children[inp][output] = new_node
581                curr_node = new_node
