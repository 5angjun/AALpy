130        shared_keys = set(c1_out_freq.keys()).intersection(c2_out_freq.keys())
131        if len(shared_keys) == 0:
132            # if the supports of the tested frequencies are completely then chi2 makes no sense, use the Hoeffding test
133            # to determine if there are enough observations for a difference
134            hoeffding_checker = AdvancedHoeffdingChecker()

138        if dof not in self.chi2_values.keys():

140        else:
141            chi2_val = self.chi2_values[dof]
142

163    def compute_Q(self, c1_out_freq, c2_out_freq, keys):
164        n_1 = sum(c1_out_freq.values())
165        n_2 = sum(c2_out_freq.values())
166
167        Q = 0
168        default_val = 0
169        yates_correction = -0.5 if len(keys) == 2 and \
170                                   any(c1_out_freq.get(k, 0) < 5 or c2_out_freq.get(k, 0) < 5 for k in keys) else 0
171        for k in keys:

173            q_1_k = float(((abs(c1_out_freq.get(k, default_val) - n_1 * p_hat_k)) + yates_correction) ** 2) / (
174                    n_1 * p_hat_k)
